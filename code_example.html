using System.Collections;
using System.Collections.Generic;
using UnityEngine;
using UnityEngine.UI;

public class ValueMenu : MonoBehaviour {

    public Dropdown xDropdown;
    public Dropdown yDropdown;
    public Dropdown zDropdown;

    public Button plotIt;


    public bool newPlot;
    public string selected_xValue;
    public string selected_yValue;
    public string selected_zValue;

    List<string> valueNames = new List<string>(){"Empty Axis" ,"Rank", "Name", "Platform", "Year", "Genre", "Publisher", "NA Sales", "EU Sales", "JP Sales", "Other Sales", "Global Sales"};


	// Use this for initialization
	void Start () {
        FillDropdown();
        selected_xValue = valueNames[0];
        selected_yValue = valueNames[0];
        selected_zValue = valueNames[0];
	}

    //The GameObject Dropdown is returning an index. Dropdown_INdexChange set the public string selected Value to the associated string
    public void XDropdown_IndexChange(int index)
    {
        selected_xValue = valueNames[index];
        xDropdown.interactable = false;
        Debug.Log("NEW xVALUE" + selected_xValue);
        Debug.Log(selected_xValue);
    }

    public void YDropdown_IndexChange(int index)
    {
        selected_yValue = valueNames[index];
        yDropdown.interactable = false;
        Debug.Log("NEW yVALUE" + selected_yValue);
    }

    public void ZDropdown_IndexChange(int index)
    {
        selected_zValue = valueNames[index];
        zDropdown.interactable = false;
        Debug.Log("NEW zVALUE" + selected_zValue);
    }

    // set newPlot true, when Button in pressed
    public void ButtonPress()
    {
        newPlot = true;
        xDropdown.interactable = true;
        yDropdown.interactable = true;
        zDropdown.interactable = true;
        Debug.Log("New Plot");

    }

    // Add each value name as an option to each dropdown
    void FillDropdown()
    {
        xDropdown.AddOptions(valueNames);
        yDropdown.AddOptions(valueNames);
        zDropdown.AddOptions(valueNames);
    }
}
